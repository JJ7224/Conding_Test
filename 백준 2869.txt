# 백준 2869 - 달팽이는 올라가고 싶다

# 내가 한 것(답은 맞음, 시간에러)
a, b, v = map(int, input().split(' '))
day = 0
i = 0

while(i < v):
    i = day + a - b
    if i < v:
        day += 1
    elif i >= v:
        print(day)
        break
        
print('-----------------------------------------')        

# 문제 풀이
a, b, v = map(int, input().split(' '))
day = 0

# 하루에 떨어졌다 올라오는 높이
d = a - b

# 마지막 날에 도착하면 되는 높이
p = v - b

# p까지 도달하는데 걸리는 날짜
day = p // d

if p % d > 0:
    day += 1
    
print(day)
